Usage

Markup

<div class="switchers" data-mode="accordion">
  <div class="switcher-head">
    <button class="switcher-toggle-all">switcher-toggle-all</button>
  </div>
  <div class="switcher">
    <div class="switcher-header">
      <div class="switcher-header-col">
        switcher-header-col
      </div>
      <div class="switcher-header-col switcher-toggle">
        switcher-header-col
      </div>
    </div>
    <div class="switcher-body">
      switcher-body
    </div>
  </div><!-- /.switcher -->
  <div class="switcher">
    ...
  </div><!-- /.switcher -->
</div>

JavaScript

$('.switchers').switchers({
  switcher:   'switcher',
  toggle:     'switcher-toggle',
  toggleAll:  'switcher-toggle-all',
  open:       'switcher-open',
  close:      'switcher-close'
});

Methods

$().switchers(options)

$().switchers('open')

$('#element').switchers('open')

$().switchers('close')

$('#element').switchers('close')

$().switchers('toggle')

$('#element').switchers('toggle')

$().switchers('toggleAll')

$('#element').switchers('toggleAll')

$().switchers('destroy')

$('#element').switchers('destroy')
