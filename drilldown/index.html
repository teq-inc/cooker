Usage

Markup

<table class="drilldown">
  <thead>
    <tr>
      <td>
        <button class="btn btn-default drilldown-open-all">All Open</button>
        <button class="btn btn-default drilldown-close-all">All Close</button>
      </td>
    </tr>
    <tr>
      <th>....</th>
    </tr>
  </thead>
  <tbody>
    <tr class="drilldown-body drilldown-toggle" data-level="1">
      <td>...</td>
    </tr>
    <tr class="drilldown-body drilldown-toggle" data-level="2" style="display:none;">
      <td>...</td>
    </tr>
    <tr class="drilldown-body drilldown-toggle" data-level="3" style="display:none;">
      <td>...</td>
    </tr>
    <tr class="drilldown-body" data-level="4" style="display:none;">
      <td>...</td>
    </tr>
    <tr class="drilldown-body drilldown-toggle" data-level="1">
      <td>...</td>
    </tr>
    <tr class="drilldown-body drilldown-toggle" data-level="2" style="display:none;">
      <td>...</td>
    </tr>
    <tr class="drilldown-body" data-level="3" style="display:none;">
      <td>...</td>
    </tr>
  </tbody>
</table>

JavaScript

$('.drilldown').drilldown({
  speed: 'fast'
});

Methods

$().drilldown(options)

$().drilldown('openAll')

$('#element').drilldown('openAll')

$().drilldown('closeAll')

$('#element').drilldown('closeAll')

$().switchers('destroy')

$('#element').drilldown('destroy')
